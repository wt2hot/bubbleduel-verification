function generateDeterministicRandom(serverSeed, clientSeed) {
    // This hash function is a simple example. For production, use a cryptographic hash like SHA-256.
    const combinedString = `${serverSeed}-${clientSeed}`;
    let hash = 0;
    for (let i = 0; i < combinedString.length; i++) {
        const char = combinedString.charCodeAt(i);
        hash = ((hash << 5) - hash) + char;
        hash |= 0; // Convert to 32bit integer
    }
    // Normalize to a value between 0 and 100
    return Math.abs(hash % 100000) / 1000; // Scale to 0-100 with decimals
}

function calculateFairness() {
    const serverSeed = document.getElementById('serverSeed').value;
    const clientSeed = document.getElementById('clientSeed').value;
    const playerA = parseFloat(playerABetInput.value);
    const playerB = parseFloat(playerBBetInput.value);

    // Validate for empty fields
    if (serverSeed.trim() === '' || clientSeed.trim() === '' || isNaN(playerA) || isNaN(playerB)) {
        chanceAOutput.textContent = 'Player A Chance: Please fill in all fields.';
        chanceBOutput.textContent = 'Player B Chance: Please fill in all fields.';
        loadingSpinner.classList.add('hidden');
        return;
    }

    // Validate seed format (alphanumeric and hyphens only)
    const seedRegex = /^[a-zA-Z0-9-]+$/;
    if (!seedRegex.test(serverSeed)) {
        chanceAOutput.textContent = 'Player A Chance: Server Seed must be alphanumeric or hyphens only.';
        chanceBOutput.textContent = 'Player B Chance: Invalid Server Seed.';
        loadingSpinner.classList.add('hidden');
        return;
    }
    if (!seedRegex.test(clientSeed)) {
        chanceAOutput.textContent = 'Player A Chance: Client Seed must be alphanumeric or hyphens only.';
        chanceBOutput.textContent = 'Player B Chance: Invalid Client Seed.';
        loadingSpinner.classList.add('hidden');
        return;
    }

    // Validate seed length
    if (serverSeed.length < 32 || serverSeed.length > 64) {
        chanceAOutput.textContent = 'Player A Chance: Server Seed must be 32-64 characters.';
        chanceBOutput.textContent = 'Player B Chance: Invalid Server Seed length.';
        loadingSpinner.classList.add('hidden');
        return;
    }
    if (clientSeed.length < 8 || clientSeed.length > 32) {
        chanceAOutput.textContent = 'Player A Chance: Client Seed must be 8-32 characters.';
        chanceBOutput.textContent = 'Player B Chance: Invalid Client Seed length.';
        loadingSpinner.classList.add('hidden');
        return;
    }

    // Validate bet amounts
    if (playerA < 0 || playerB < 0) {
        chanceAOutput.textContent = 'Player A Chance: Bet amounts cannot be negative.';
        chanceBOutput.textContent = 'Player B Chance: Bet amounts cannot be negative.';
        loadingSpinner.classList.add('hidden');
        return;
    }

    if (playerA === 0 && playerB === 0) {
        chanceAOutput.textContent = 'Player A Chance: At least one bet must be greater than 0.';
        chanceBOutput.textContent = 'Player B Chance: At least one bet must be greater than 0.';
        loadingSpinner.classList.add('hidden');
        return;
    }


    const total = playerA + playerB;
    let chanceA = (playerA / total) * 100;
    let chanceB = (playerB / total) * 100;

    if (playerA === 0 && playerB > 0) {
        chanceA = 0;
        chanceB = 100;
    } else if (playerB === 0 && playerA > 0) {
        chanceA = 100;
        chanceB = 0;
    } else if (playerA === 0 && playerB === 0) {
        chanceA = 50;
        chanceB = 50;
    }

    const derivedRand = generateDeterministicRandom(serverSeed, clientSeed);
    let winnerText = '';

    if (derivedRand < chanceA) {
        winnerText = 'Player A wins!';
    } else {
        winnerText = 'Player B wins!';
    }

    chanceAOutput.textContent = `Player A Chance: ${chanceA.toFixed(2)}%`;
    chanceBOutput.textContent = `Player B Chance: ${chanceB.toFixed(2)}%`;
}
